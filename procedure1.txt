CREATE OR REPLACE PROCEDURE simple_select()
  RETURNS STRING
  LANGUAGE JAVASCRIPT
AS
$$
  var sql = "SELECT * FROM my_table"; // 查询语句
  var result = snowflake.execute(sql); // 执行查询
  var output = '';

  // 遍历查询结果并拼接为字符串
  while (result.next()) {
    output += result.getColumnValue(1) + ', ' + result.getColumnValue(2) + '\n';
  }

  return output; // 返回结果
$$;
